    **Step 1: Create docker-compose.yml**
    Per D-12, D-14, and RESEARCH.md Pitfall 5:
    ```yaml
    services:
      postgres:
        image: postgres:16-alpine
        environment:
          POSTGRES_DB: finance_manager
          POSTGRES_USER: postgres
          POSTGRES_PASSWORD: postgres
        ports:
          - "5432:5432"
        volumes:
          - pgdata:/var/lib/postgresql/data
        healthcheck:
          test: ["CMD-SHELL", "pg_isready -U postgres"]
          interval: 5s
          timeout: 5s
          retries: 5

      backend:
        build:
          context: ./backend
          dockerfile: Dockerfile
        ports:
          - "3000:3000"
        environment:
          DB_HOST: postgres
          DB_PORT: 5432
          DB_USERNAME: postgres
          DB_PASSWORD: postgres
          DB_NAME: finance_manager
          PORT: 3000
        depends_on:
          postgres:
            condition: service_healthy
        healthcheck:
          test: ["CMD", "curl", "-f", "http://localhost:3000/api/health"]
          interval: 10s
          timeout: 5s
          retries: 5
          start_period: 30s

      frontend:
        build:
          context: ./frontend
          dockerfile: Dockerfile
        ports:
          - "8080:80"
        depends_on:
          backend:
            condition: service_healthy

    volumes:
      pgdata:
    ```

    Key: `depends_on` with `condition: service_healthy` prevents backend from starting until PostgreSQL is ready (Pitfall 5).

    **Step 2: Create backend/Dockerfile**
    Multi-stage build using Node 22 (matches PROJECT.md specification, not local Node 24):
    ```dockerfile
    FROM node:22-alpine AS builder
    WORKDIR /app
    COPY package*.json ./
    RUN npm ci
    COPY . .
    RUN npm run build

    FROM node:22-alpine AS runner
    WORKDIR /app
    COPY package*.json ./
    RUN npm ci --only=production
    COPY --from=builder /app/dist ./dist
    COPY --from=builder /app/migrations ./migrations
    COPY --from=builder /app/data-source.ts ./data-source.ts
    COPY --from=builder /app/tsconfig.json ./tsconfig.json

    # Install curl for healthcheck and ts-node for migrations
    RUN apk add --no-cache curl

    EXPOSE 3000

    # Run migrations then start app
    CMD ["sh", "-c", "npx typeorm migration:run -d data-source.ts && node dist/main.js"]
    ```

    Note: The CMD runs migrations before starting the app, ensuring the schema exists.

    **Step 3: Create frontend/Dockerfile**
    Multi-stage build: build with Vite, serve with nginx:
    ```dockerfile
    FROM node:22-alpine AS builder
    WORKDIR /app
    COPY package*.json ./
    RUN npm ci
    COPY . .
    RUN npm run build

    FROM nginx:alpine
    COPY --from=builder /app/dist /usr/share/nginx/html
    COPY nginx.conf /etc/nginx/conf.d/default.conf
    EXPOSE 80
    CMD ["nginx", "-g", "daemon off;"]
    ```

    **Step 4: Create frontend/nginx.conf**
    Nginx config that serves the SPA and proxies API requests to the backend:
    ```nginx
    server {
        listen 80;
        server_name localhost;
        root /usr/share/nginx/html;
        index index.html;

        location /api {
            proxy_pass http://backend:3000;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        location / {
            try_files $uri $uri/ /index.html;
        }
    }
    ```
    This replaces the Vite proxy in production -- nginx proxies /api to the backend container.
