    **Step 2: Create frontend/src/api/client.ts**
    API client using native fetch (no axios dependency needed):
    ```typescript
    const API_BASE = '/api';

    export async function apiClient<T>(
      endpoint: string,
      options?: RequestInit,
    ): Promise<T> {
      const response = await fetch(`${API_BASE}${endpoint}`, {
        headers: {
          'Content-Type': 'application/json',
          ...options?.headers,
        },
        ...options,
      });

      if (!response.ok) {
        const error = await response.json().catch(() => ({ message: 'Request failed' }));
        throw new Error(error.message || `HTTP ${response.status}`);
      }

      if (response.status === 204) return undefined as T;
      return response.json();
    }
    ```

    **Step 3: Create frontend/src/api/assets.ts**
    Asset API functions using the client:
    ```typescript
    import { apiClient } from './client';
    import type { Asset, CreateAssetRequest, UpdateAssetRequest } from '../types/api';

    export const assetsApi = {
      getAll: (category?: string): Promise<Asset[]> => {
        const params = category ? `?category=${category}` : '';
        return apiClient<Asset[]>(`/assets${params}`);
      },

      create: (data: CreateAssetRequest): Promise<Asset> =>
        apiClient<Asset>('/assets', {
          method: 'POST',
          body: JSON.stringify(data),
        }),

      update: (id: string, data: UpdateAssetRequest): Promise<Asset> =>
        apiClient<Asset>(`/assets/${id}`, {
          method: 'PUT',
          body: JSON.stringify(data),
        }),

      remove: (id: string): Promise<void> =>
        apiClient<void>(`/assets/${id}`, {
          method: 'DELETE',
        }),
    };
    ```
